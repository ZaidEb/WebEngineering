Rekursion

Draws a circle, then calls itself twice to draw 2 smaller circles below.

Enter size (e.g. 200):





Rekursion output:

function drawCircles(left, top, size) {
  if (size > 10) {
    // Draw circle
    ctx.beginPath();
    ctx.arc(left+size/2, top+size/2, size/2, 0, Math.PI*2);
    ctx.stroke();
    
    // Recursion
    let s = size / 2;
    let newTop = top + s / 2;
    
    drawCircles(left, newTop, s);
    drawCircles(left + s, newTop, s);
  }
}

  if (size >= 16) {
    // Draw square
    ctx.fillStyle = "#ccc";
    ctx.fillRect(left, top, size, size);
    ctx.strokeStyle = "#000";
    ctx.strokeRect(left, top, size, size);
    
    // Recursion (4 corners)
    let s = size / 2 - margin;
    drawSquares(left - s/2, top - s/2, s);
    drawSquares(left + size - s/2, top - s/2, s);
    drawSquares(left - s/2, top + size - s/2, s);
    drawSquares(left + size - s/2, top + size - s/2, s);
  }
}
